78inline Arr matmul(const Arr& A, const Arr& B) {
79 assert(A.is_2d() && B.is_2d() && A.cols() == B.rows());
80 auto m = A.rows();
81 auto k = A.cols();
82 auto n = B.cols();
83 Arr out(m, n);
84 for (size_t i = 0; i < m; ++i)
85 for (size_t j = 0; j < n; ++j) {
86 double s = 0.0;
87 for (size_t t = 0; t < k; ++t) s += A(i, t) * B(t, j);
88 out(i, j) = s;
89 }
90 return out;

94// Cholesky decomposition: A = L * L^T (A symmetric positive definite)
95// Returns lower-triangular L.
96// ---------------------------------------------------------------------------
97
103inline Arr cholesky(const Arr& A) {
104 auto n = A.rows();
105 assert(A.is_2d() && A.cols() == n);
106 Arr L(n, n);
107 for (size_t j = 0; j < n; ++j) {
108 double s = 0.0;
109 for (size_t k = 0; k < j; ++k) s += L(j, k) * L(j, k);
110 L(j, j) = std::sqrt(A(j, j) - s);
111 for (size_t i = j + 1; i < n; ++i) {
112 s = 0.0;
113 for (size_t k = 0; k < j; ++k) s += L(i, k) * L(j, k);
114 L(i, j) = (A(i, j) - s) / L(j, j);
115 }
116 }
117 return L;
118}

129inline Arr inv(const Arr& A) {
130 auto n = A.rows();
131 auto L = cholesky(A);
132 // Solve L * Y = I (forward substitution)
133 Arr Y(n, n);
134 for (size_t j = 0; j < n; ++j) {
135 for (size_t i = 0; i < n; ++i) {
136 double s = (i == j) ? 1.0 : 0.0;
137 for (size_t k = 0; k < i; ++k) s -= L(i, k) * Y(k, j);
138 Y(i, j) = s / L(i, i);
139 }
140 }
141 // Solve L^T * X = Y (back substitution)
142 Arr X(n, n);
143 for (size_t j = 0; j < n; ++j) {
144 for (size_t i = n; i-- > 0;) {
145 double s = Y(i, j);
146 for (size_t k = i + 1; k < n; ++k) s -= L(k, i) * X(k, j);
147 X(i, j) = s / L(i, i);
148 }
149 }
150 return X;
151}
